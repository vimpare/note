var longestCommonPrefix = function(strs) {
   var first = strs[0]
   if(!strs.length){
       return ''
   }
    var str = '';
    for (var i = 0; i < first.length; i++) {
        for (var j = 1; j < strs.length; j++) {
            if (first[i] != strs[j][i]) {
                return str;
            }
        }
        str += first[i];
    }
    return str 
};

    let candidate = 0;

var isAnagram = function (s, t) {
    if (s.length != t.length) return false;
    for (var i = 0; i < s.length; i++) {
        var str = s[i]
        if (t.indexOf(str)!=-1) {
            t = t.replace(str, '')
        }
    }
    return t?false:true
};
